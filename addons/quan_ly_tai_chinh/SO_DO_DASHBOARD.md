## 6. SƠ ĐỒ TÍNH TOÁN

```
┌─────────────────────────────────────────────────────────────┐
│            CÔNG THỨC TÍNH TOÁN CHỈ SỐ                       │
└─────────────────────────────────────────────────────────────┘

┌──────────────────────────────────────────────────────────────┐
│  1. ĐỀ XUẤT MUA THIẾT BỊ                                     │
├──────────────────────────────────────────────────────────────┤
│                                                              │
│  tong_de_xuat = COUNT(de_xuat_mua_thiet_bi)                 │
│                                                              │
│  de_xuat_cho_duyet = COUNT(de_xuat_mua_thiet_bi             │
│                            WHERE state = 'submitted')        │
│                                                              │
│  tong_gia_tri_de_xuat = SUM(de_xuat_mua_thiet_bi            │
│                             .tong_gia_tri)                   │
│                                                              │
└──────────────────────────────────────────────────────────────┘

┌──────────────────────────────────────────────────────────────┐
│  2. TÀI SẢN                                                  │
├──────────────────────────────────────────────────────────────┤
│                                                              │
│  tong_tai_san = COUNT(khau_hao)                             │
│                                                              │
│  tai_san_hoat_dong = COUNT(khau_hao                         │
│                            WHERE state = 'active')           │
│                                                              │
│  tong_gia_tri_tai_san = SUM(khau_hao.gia_tri)               │
│                                                              │
│  tong_khau_hao_tich_luy = SUM(khau_hao                      │
│                               .accumulated_depreciation)     │
│                                                              │
│  tong_gia_tri_con_lai = tong_gia_tri_tai_san -              │
│                         tong_khau_hao_tich_luy               │
│                                                              │
└──────────────────────────────────────────────────────────────┘

┌──────────────────────────────────────────────────────────────┐
│  3. KHẤU HAO THEO KỲ                                         │
├──────────────────────────────────────────────────────────────┤
│                                                              │
│  today = Date.today()                                        │
│  start_month = today.replace(day=1)                          │
│  end_month = start_month + relativedelta(months=1, days=-1) │
│                                                              │
│  khau_hao_thang_nay = SUM(khau_hao.line.amount              │
│                           WHERE date >= start_month          │
│                           AND date <= end_month              │
│                           AND state = 'depreciated')         │
│                                                              │
│  ┌────────────────────────────────────────────────────────┐ │
│  │  Tương tự cho:                                         │ │
│  │  • khau_hao_quy_nay                                    │ │
│  │  • khau_hao_nam_nay                                    │ │
│  └────────────────────────────────────────────────────────┘ │
│                                                              │
└──────────────────────────────────────────────────────────────┘

┌──────────────────────────────────────────────────────────────┐
│  4. XU HƯỚNG 12 THÁNG                                        │
├──────────────────────────────────────────────────────────────┤
│                                                              │
│  FOR i IN range(11, -1, -1):                                │
│      start_date = (today - relativedelta(months=i))         │
│                   .replace(day=1)                            │
│      end_date = start_date + relativedelta(months=1,        │
│                                            days=-1)          │
│                                                              │
│      trend_amount = SUM(khau_hao.line.amount                │
│                        WHERE date >= start_date              │
│                        AND date <= end_date                  │
│                        AND state = 'depreciated')            │
│                                                              │
│      CREATE dashboard.depreciation.trend(                    │
│          month = start_date.strftime('%m/%Y'),              │
│          amount = trend_amount                              │
│      )                                                       │
│                                                              │
└──────────────────────────────────────────────────────────────┘

┌──────────────────────────────────────────────────────────────┐
│  5. CẢNH BÁO                                                 │
├──────────────────────────────────────────────────────────────┤
│                                                              │
│  canh_bao_tai_san_sap_khau_hao_het =                        │
│      COUNT(khau_hao                                         │
│            WHERE (gia_tri_con_lai / gia_tri) * 100 < 10     │
│            AND state = 'active')                             │
│                                                              │
│  canh_bao_de_xuat_qua_han =                                 │
│      COUNT(de_xuat_mua_thiet_bi                             │
│            WHERE ngay_du_kien_nhan < today                  │
│            AND state IN ('submitted', 'approved'))           │
│                                                              │
└──────────────────────────────────────────────────────────────┘
```
